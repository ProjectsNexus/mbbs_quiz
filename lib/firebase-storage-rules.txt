rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {

    // Notes PDFs - stored at notes/{year}/{block}/{subject}/{topic}/{file}
    // Any authenticated user can view notes; only authenticated users can upload (tutors via app logic).
    match /notes/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null
                   && request.resource.contentType == 'application/pdf'
                   && request.resource.size < 25 * 1024 * 1024; // 25MB limit
      allow delete: if request.auth != null;
    }
  }
}
